#### fileioLab1.py

~~~python
import os                                                         
import datetime                                                   
                                                                  
t = ['월', '화', '수', '목', '금', '토', '일']                           
d = datetime.datetime.today()                                     
r = d.weekday()                                                   
d = str(d)                                                        
m = datetime.datetime(1995, 5, 2).weekday()                       
try:                                                              
    f = open('c:/iotest/today.txt', 'wt')                         
except:                                                           
    file = 'c:/iotest'                                            
    if not os.path.isdir(file):                                   
        os.mkdir(file)                                            
        f = open('c:/iotest/today.txt', 'wt')                     
finally:                                                          
    f.write("오늘은 %s년 %s월 %s일입니다.\n"%(d[0:4], d[5:7], d[8:10]))    
    f.write("오늘은 %s요일입니다.\n" %t[r])                               
    f.write("나는 %s요일에 태어났습니다.\n" %t[m])                           
    print("저장이 완료되었습니다.")                                         
    f.close()                                                     
~~~



#### fileioLab2.py

~~~python
try:
    f1 = open('sample.txt', 'r', encoding='UTF8')
    f2 = open('sample_2021_01_18.txt', 'a')
    rows = f1.readlines()
    for row in rows:
        f2.write(row)
    print("저장이 완료되었습니다.")
except FileNotFoundError as e:
    print(e)
finally:
    f1.close()
    f2.close()
~~~

#### stLab1.py

~~~python
import calendar

while True:
    try:
        year, month = map(int, input("연도와 월을 입력하세요. : ").split())
        if year <= 0 or 0 >= month or 12 < month:
            raise ValueError
        break
    except ValueError:
        print("잘 못 입력하셨습니다.")
print(year, month)
calendar.prmonth(year, month)
~~~
