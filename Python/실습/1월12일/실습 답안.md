#### funcLab11

~~~ python
def mydict(**kwargs):
    result = dict()
    if len(kwargs) > 0:
        for i, j in kwargs.items():
            result['my'+i] = j
        return result
    else:
        return result
print(mydict(a=1, b=2, c=3))
print(mydict())
print(mydict(교육='멀티캠퍼스',현재='파이썬'))
~~~



####  funcLab12

~~~python
def myprint(*args, **kwargs):
    deco = "**"
    sep = ","
    if len(args) > 0:
        if 'deco' in kwargs:
            deco = kwargs['deco']
        if 'sep' in kwargs:
            sep = kwargs['sep']
        result = [str(i) for i in args]
        joined_str = sep.join(result)
        print(deco, joined_str, deco, sep = "")
    else:
        print("Hello Python")

myprint(10, 20, 30, deco="@", sep="-")
myprint("python", "javascript", "R", deco="$")
myprint("가", "나", "다")
myprint(100)
myprint(True, 111, False, "abc", deco="&", sep="")
myprint()
~~~
